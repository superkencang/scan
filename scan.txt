// Helper to calculate Shannon Entropy in PHP
function calculate_entropy($text) {
    if (!$text || strlen($text) === 0) return 0;
    $len = strlen($text);
    $freqs = [];

    for ($i = 0; $i < $len; $i++) {
        $char = $text[$i];
        if (!isset($freqs[$char])) {
            $freqs[$char] = 0;
        }
        $freqs[$char]++;
    }

    $entropy = 0;
    foreach ($freqs as $count) {
        $p = $count / $len;
        $entropy -= $p * log($p, 2);
    }

    return round($entropy, 3);
}

// Scans code text content
function scan_code($content, $filename, $rules) {
    $entropy = calculate_entropy($content);
    $lines = explode("\n", $content);
    $matches = [];
    $categories = [];

    foreach ($lines as $index => $line) {
        $lineNumber = $index + 1;
        $truncatedLine = strlen($line) > 500 ? substr($line, 0, 500) . " [TRUNCATED]" : $line;

        foreach ($rules as $rule) {
            if (preg_match($rule['pattern'], $line)) {
                $matches[] = [
                    "line" => $lineNumber,
                    "content" => trim($truncatedLine),
                    "ruleName" => $rule['name'],
                    "description" => $rule['description'],
                    "severity" => $rule['severity']
                ];
                $categories[$rule['category']] = true;
            }
        }

        // Long Line Base64/Hex indicator
        if (preg_match('/[a-zA-Z0-9+\/=]{150,}/', $line, $longMatch)) {
            $matches[] = [
                "line" => $lineNumber,
                "content" => substr($longMatch[0], 0, 100) . "...",
                "ruleName" => "Extremely Long Continuous String",
                "description" => "Detects continuous characters of length > 150 without spaces, often signaling raw base64 or hex code injection.",
                "severity" => "medium"
            ];
            $categories["Obfuscation"] = true;
        }
    }

    $isObfuscated = $entropy > 5.8 && (count($categories) > 0 || strpos(serialize($matches), 'Hex') !== false);
    if ($isObfuscated) {
        $categories["Obfuscation"] = true;
    }

    // Determine overall risk score
    $baseScore = 0;
    foreach ($matches as $m) {
        if ($m['severity'] == "critical") $baseScore += 45;
        elseif ($m['severity'] == "high") $baseScore += 25;
        elseif ($m['severity'] == "medium") $baseScore += 10;
        else $baseScore += 3;
    }

    if ($entropy > 6.0) $baseScore += 20;
    elseif ($entropy > 5.4) $baseScore += 10;

    $riskScore = min(100, max(0, $baseScore));

    $riskLevel = "safe";
    if ($riskScore >= 75) $riskLevel = "critical";
    elseif ($riskScore >= 45) $riskLevel = "high";
    elseif ($riskScore >= 20) $riskLevel = "medium";
    elseif ($riskScore > 0) $riskLevel = "low";

    return [
        "filePath" => $filename,
        "fileSize" => strlen($content),
        "entropy" => $entropy,
        "riskScore" => $riskScore,
        "riskLevel" => $riskLevel,
        "matches" => $matches,
        "isObfuscated" => $isObfuscated,
        "categories" => array_keys($categories)
    ];
}

// Scans recursively down folders
function scan_directory_recursive($dir, $rules, &$results, $extensions) {
    if (!is_dir($dir)) return;
    $files = scandir($dir);
    if (!$files) return;

    foreach ($files as $file) {
        if ($file === '.' || $file === '..') continue;
        
        $path = $dir . DIRECTORY_SEPARATOR . $file;
        
        // Skip ignored directories
        $ignoredDirs = ['node_modules', '.git', 'dist', 'build', '.next', 'cache'];
        if (is_dir($path)) {
            if (in_array($file, $ignoredDirs)) continue;
            scan_directory_recursive($path, $rules, $results, $extensions);
        } else {
            $ext = strtolower(pathinfo($path, PATHINFO_EXTENSION));
            if (in_array('.' . $ext, $extensions) || in_array($ext, ['php', 'phtml', 'html', 'js', 'py', 'sh', 'pl'])) {
                $content = @file_get_contents($path);
                if ($content !== false) {
                    $scanResult = scan_code($content, $path, $rules);
                    if ($scanResult['riskScore'] > 0 || count($scanResult['matches']) > 0) {
                        $results[] = $scanResult;
                    }
                }
            }
        }
    }
}

// Handle Get Directory Contents for File Explorer
if (isset($_GET['action']) && $_GET['action'] == 'explore_dir' && isset($_GET['path'])) {
    $path = $_GET['path'];
    $realPath = realpath($path);
    
    if ($realPath && is_dir($realPath)) {
        $items = [];
        $files = scandir($realPath);
        foreach ($files as $item) {
            if ($item === '.') continue;
            $fullPath = $realPath . DIRECTORY_SEPARATOR . $item;
            $isDir = is_dir($fullPath);
            $items[] = [
                'name' => $item,
                'is_dir' => $isDir,
                'path' => $fullPath,
                'size' => $isDir ? 0 : filesize($fullPath),
                'modified' => filemtime($fullPath)
            ];
        }
        usort($items, function($a, $b) {
            if ($a['is_dir'] && !$b['is_dir']) return -1;
            if (!$a['is_dir'] && $b['is_dir']) return 1;
            return strcasecmp($a['name'], $b['name']);
        });
        
        $parts = explode('/', trim($realPath, '/'));
        $breadcrumb = [];
        $cumulative = '';
        foreach ($parts as $part) {
            if (empty($part)) continue;
            $cumulative .= '/' . $part;
            $breadcrumb[] = ['name' => $part, 'path' => $cumulative];
        }
        
        header('Content-Type: application/json');
        echo json_encode([
            'success' => true,
            'current_path' => $realPath,
            'parent' => dirname($realPath),
            'items' => $items,
            'breadcrumb' => $breadcrumb,
            'is_root' => ($realPath === '/')
        ]);
    } else {
        header('Content-Type: application/json');
        echo json_encode([
            'success' => false,
            'error' => 'Invalid directory path'
        ]);
    }
    exit;
}

<script>
    let currentFile = null;
    let currentPath = '<?php echo addslashes($scanPath); ?>';
    let navigationHistory = [currentPath];
    let historyIndex = 0;
    let fileContent = '';

    // ============ NAVIGATION ============
    
    function navigateTo(path) {
        if (path === currentPath) return;
        
        navigationHistory = navigationHistory.slice(0, historyIndex + 1);
        navigationHistory.push(path);
        historyIndex = navigationHistory.length - 1;
        currentPath = path;
        
        document.getElementById('scanPathInput').value = path;
        document.getElementById('pathDisplay').textContent = path;
        
        updateNavButtons();
        loadExplorer(path);
        showToast('Navigated to: ' + path);
    }

    function navigateBack() {
        if (historyIndex > 0) {
            historyIndex--;
            currentPath = navigationHistory[historyIndex];
            document.getElementById('scanPathInput').value = currentPath;
            document.getElementById('pathDisplay').textContent = currentPath;
            updateNavButtons();
            loadExplorer(currentPath);
        }
    }

    function navigateForward() {
        if (historyIndex < navigationHistory.length - 1) {
            historyIndex++;
            currentPath = navigationHistory[historyIndex];
            document.getElementById('scanPathInput').value = currentPath;
            document.getElementById('pathDisplay').textContent = currentPath;
            updateNavButtons();
            loadExplorer(currentPath);
        }
    }

    function refreshExplorer() {
        loadExplorer(currentPath);
        showToast('Refreshed');
    }

    function updateNavButtons() {
        document.getElementById('backBtn').disabled = (historyIndex <= 0);
        document.getElementById('forwardBtn').disabled = (historyIndex >= navigationHistory.length - 1);
    }

    // ============ FILE EXPLORER ============
    
    function loadExplorer(path) {
        const contents = document.getElementById('dirContents');
        contents.innerHTML = '<div class="dir-loading">Loading directory contents...</div>';
        document.getElementById('breadcrumbContainer').innerHTML = 
            '<span style="font-size:11px;color:#666;font-weight:600;margin-right:4px;">📂</span>' +
            '<span class="breadcrumb-item" onclick="navigateTo(\'/\')" style="font-weight:600;">Root</span>' +
            '<span class="breadcrumb-separator">›</span>' +
            '<span style="font-size:12px;color:#999;">Loading...</span>';
        
        fetch('?action=explore_dir&path=' + encodeURIComponent(path))
            .then(res => res.json())
            .then(data => {
                if (!data.success) {
                    contents.innerHTML = '<div class="dir-loading" style="color:#cc0000;">Error: ' + (data.error || 'Invalid directory') + '</div>';
                    return;
                }
                
                let breadcrumbHtml = '';
                breadcrumbHtml += '<span style="font-size:11px;color:#666;font-weight:600;margin-right:4px;">📂</span>';
                
                if (data.is_root) {
                    breadcrumbHtml += '<span class="breadcrumb-item current" style="font-weight:600;">/</span>';
                } else {
                    breadcrumbHtml += '<span class="breadcrumb-item" onclick="navigateTo(\'/\')" style="font-weight:600;">Root</span>';
                    
                    const parts = data.current_path.split('/').filter(p => p.length > 0);
                    let cumulative = '';
                    parts.forEach((part, index) => {
                        cumulative += '/' + part;
                        const isLast = index === parts.length - 1;
                        breadcrumbHtml += '<span class="breadcrumb-separator">›</span>';
                        breadcrumbHtml += `<span class="breadcrumb-item ${isLast ? 'current' : ''}" onclick="navigateTo('${cumulative}')">${part}</span>`;
                    });
                }
                
                document.getElementById('breadcrumbContainer').innerHTML = breadcrumbHtml;
                
                let contentsHtml = '';
                
                if (data.parent && data.parent !== data.current_path && data.parent !== '') {
                    contentsHtml += `<div class="dir-item parent" onclick="navigateTo('${data.parent}')">
                        <span class="icon">📂</span>
                        <span class="name">..</span>
                    </div>`;
                }
                
                if (data.items && data.items.length > 0) {
                    let hasItems = false;
                    data.items.forEach(item => {
                        if (item.name === '..') return;
                        hasItems = true;
                        const icon = item.is_dir ? '📁' : '📄';
                        const sizeText = item.is_dir ? '' : (item.size > 0 ? (item.size / 1024).toFixed(1) + ' KB' : '');
                        const onclick = item.is_dir ? `navigateTo('${item.path}')` : '';
                        const cls = item.is_dir ? '' : 'file-item-dir';
                        contentsHtml += `<div class="dir-item ${cls}" onclick="${onclick}">
                            <span class="icon">${icon}</span>
                            <span class="name">${item.name}</span>
                            <span class="size">${sizeText}</span>
                        </div>`;
                    });
                    if (!hasItems) {
                        contentsHtml = '<div class="dir-empty">Empty directory</div>';
                    }
                } else {
                    contentsHtml = '<div class="dir-empty">Empty directory</div>';
                }
                
                contents.innerHTML = contentsHtml;
                document.getElementById('pathDisplay').textContent = data.current_path;
            })
            .catch(() => {
                contents.innerHTML = '<div class="dir-loading" style="color:#cc0000;">Failed to load directory</div>';
            });
    }

    // ============ BOOKMARKS ============
    
    function saveBookmark() {
        const name = document.getElementById('bookmarkName').value.trim();
        const path = document.getElementById('bookmarkPath').value.trim();
        
        if (!name || !path) {
            showToast('Please enter both name and path');
            return;
        }
        
        fetch('', {
            method: 'POST',
            headers: { 'Content-Type': 'application/x-www-form-urlencoded' },
            body: 'action=save_bookmark&name=' + encodeURIComponent(name) + '&path=' + encodeURIComponent(path)
        })
        .then(res => res.json())
        .then(data => {
            if (data.success) {
                showToast('Bookmark saved!');
                setTimeout(() => location.reload(), 800);
            } else {
                showToast('Error saving bookmark');
            }
        })
        .catch(() => showToast('Error saving bookmark'));
    }

    function deleteBookmark(index) {
        if (!confirm('Remove this bookmark?')) return;
        
        fetch('?action=delete_bookmark&index=' + index)
            .then(res => res.json())
            .then(data => {
                if (data.success) {
                    showToast('Bookmark removed');
                    setTimeout(() => location.reload(), 800);
                } else {
                    showToast('Error removing bookmark');
                }
            })
            .catch(() => showToast('Error removing bookmark'));
    }

    // ============ TOAST ============
    
    function showToast(message) {
        const toast = document.getElementById('toast');
        toast.textContent = message;
        toast.classList.add('show');
        clearTimeout(toast._timeout);
        toast._timeout = setTimeout(() => toast.classList.remove('show'), 3000);
    }

    // ============ FILE OPERATIONS ============
    
    function selectFile(fileData) {
        currentFile = fileData;
        document.getElementById('no-selection').style.display = 'none';
        const content = document.getElementById('detail-content');
        content.classList.remove('detail-hidden');
        content.style.display = 'block';

        document.getElementById('detail-filename').textContent = fileData.filePath;
        document.getElementById('detail-filesize').textContent = (fileData.fileSize / 1024).toFixed(2) + ' KB';
        document.getElementById('detail-entropy').textContent = fileData.entropy;

        const badge = document.getElementById('detail-risk-badge');
        badge.textContent = fileData.riskLevel;
        badge.style.borderColor = '#000000';

        const matchesBox = document.getElementById('detail-matches');
        matchesBox.innerHTML = '';
        if (fileData.matches && fileData.matches.length > 0) {
            fileData.matches.forEach(m => {
                const div = document.createElement('div');
                div.className = 'match-item';
                div.innerHTML = `
                    <div class="match-header">
                        <span class="match-rule">${m.ruleName}</span>
                        <span class="match-line">Line ${m.line}</span>
                    </div>
                    <div class="match-desc">${m.description}</div>
                    <pre class="match-code">${m.content}</pre>
                `;
                matchesBox.appendChild(div);
            });
        } else {
            matchesBox.innerHTML = '<div style="padding:12px 0;color:#999;font-style:italic;font-size:11px;">No static detection rules triggered.</div>';
        }

        const codeBuffer = document.getElementById('detail-code-buffer');
        codeBuffer.textContent = 'Loading file content...';
        fetch('?action=view_file&file=' + encodeURIComponent(fileData.filePath))
            .then(res => res.text())
            .then(text => { 
                fileContent = text;
                codeBuffer.textContent = text; 
            })
            .catch(() => { 
                codeBuffer.textContent = 'Error: Failed to fetch file contents.';
                fileContent = '';
            });
    }

    // ============ VIEW FILE ============
    
    function viewFile() {
        if (!currentFile) {
            showToast('No file selected');
            return;
        }
        
        const modal = document.getElementById('viewModal');
        const content = document.getElementById('viewContent');
        
        if (fileContent) {
            content.textContent = fileContent;
        } else {
            content.textContent = 'Loading file content...';
            fetch('?action=view_file&file=' + encodeURIComponent(currentFile.filePath))
                .then(res => res.text())
                .then(text => {
                    fileContent = text;
                    content.textContent = text;
                })
                .catch(() => {
                    content.textContent = 'Error loading file content.';
                });
        }
        
        modal.classList.add('show');
    }

    function closeViewModal() {
        document.getElementById('viewModal').classList.remove('show');
    }

    // ============ COPY FILE ============
    
    function copyFile() {
        if (!currentFile) {
            showToast('No file selected');
            return;
        }
        
        const modal = document.getElementById('copyModal');
        document.getElementById('copySource').value = currentFile.filePath;
        // Suggest destination: same directory with _copy suffix
        const dir = currentFile.filePath.substring(0, currentFile.filePath.lastIndexOf('/'));
        const name = currentFile.filePath.substring(currentFile.filePath.lastIndexOf('/') + 1);
        const ext = name.includes('.') ? name.substring(name.lastIndexOf('.')) : '';
        const base = name.includes('.') ? name.substring(0, name.lastIndexOf('.')) : name;
        document.getElementById('copyDestination').value = dir + '/' + base + '_copy' + ext;
        modal.classList.add('show');
    }

    function closeCopyModal() {
        document.getElementById('copyModal').classList.remove('show');
    }

    function executeCopy() {
        const source = document.getElementById('copySource').value;
        const destination = document.getElementById('copyDestination').value.trim();
        
        if (!destination) {
            showToast('Please enter destination path');
            return;
        }
        
        const btn = document.querySelector('.modal-actions .primary');
        btn.textContent = 'Copying...';
        btn.disabled = true;
        
        fetch('', {
            method: 'POST',
            headers: { 'Content-Type': 'application/x-www-form-urlencoded' },
            body: 'action=copy_file&source=' + encodeURIComponent(source) + '&destination=' + encodeURIComponent(destination)
        })
        .then(res => res.text())
        .then(result => {
            showToast(result);
            closeCopyModal();
            setTimeout(() => location.reload(), 1500);
        })
        .catch(() => {
            showToast('Error copying file');
            btn.textContent = 'Copy';
            btn.disabled = false;
        });
    }

    // ============ DELETE FILE ============
    
    function deleteFile() {
        if (!currentFile) return;
        if (!confirm('Delete this file?\n' + currentFile.filePath)) return;
        
        const btn = document.getElementById('delete-btn');
        btn.textContent = 'Deleting...';
        btn.disabled = true;
        
        fetch('?action=delete_file&file=' + encodeURIComponent(currentFile.filePath))
            .then(res => res.text())
            .then(result => {
                showToast(result);
                setTimeout(() => location.reload(), 1500);
            })
            .catch(() => {
                showToast('Error deleting file.');
                btn.textContent = 'Delete File';
                btn.disabled = false;
            });
    }

    // ============ KEYBOARD SHORTCUTS ============
    
    document.addEventListener('keydown', function(e) {
        if (e.altKey && e.key === 'ArrowLeft') {
            e.preventDefault();
            navigateBack();
        } else if (e.altKey && e.key === 'ArrowRight') {
            e.preventDefault();
            navigateForward();
        } else if (e.key === 'Escape') {
            closeCopyModal();
            closeViewModal();
        }
    });

    // Enter key support for bookmark form
    document.getElementById('bookmarkPath').addEventListener('keypress', function(e) {
        if (e.key === 'Enter') saveBookmark();
    });

    // Close modal on outside click
    document.querySelectorAll('.modal').forEach(modal => {
        modal.addEventListener('click', function(e) {
            if (e.target === this) {
                this.classList.remove('show');
            }
        });
    });
